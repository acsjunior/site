Funções

\(f(x) = 2x\)

f <- function(x) {
  out <- 2 * x
  return(out)
}
f(2)
## [1] 4
f(4)
## [1] 8


\(g(x,y) = x^2 + y^2\)

g <- function(x, y) {
  out <- x^2 + y^2
  return(out)
}
g(1,2)
## [1] 5
g(2,3)
## [1] 13
